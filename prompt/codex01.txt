GIT 배포

# 1. 현재 변경사항 확인
git status --short
# (필요시) 변경된 requirements.txt 파일이 있으면 Git 커밋 준비 영역에 추가합니다.
git add requirements.txt

# 2. 변경 파일 스테이징
# 전체 저장소의 수정/추가/삭제 모두 스테이징
git add -A
# 현재 폴더 아래 수정/추가 중심, Git 버전에 따라 삭제 반영 차이 주의
git add .
# 추적 중인 파일의 수정/삭제만, 새 파일은 제외
git add -u

# 3. 커밋 생성
git commit -m "INIT Data Editing System - 2026.07.04-3"
# 4. 원격 저장소 최신 상태 반영
git pull origin main
# 5. 원격 저장소에 푸시
git push origin main

SELECT
    s.sid,
    s.serial#,
    s.username,
    s.status,
    s.machine,
    s.program,
    s.module,
    s.event,
    s.blocking_session,
    o.owner,
    o.object_name,
    o.object_type,
    l.locked_mode,
    CASE l.locked_mode
        WHEN 0 THEN 'NONE'
        WHEN 1 THEN 'NULL'
        WHEN 2 THEN 'ROW-S'
        WHEN 3 THEN 'ROW-X'
        WHEN 4 THEN 'SHARE'
        WHEN 5 THEN 'S/ROW-X'
        WHEN 6 THEN 'EXCLUSIVE'
        ELSE TO_CHAR(l.locked_mode)
    END AS locked_mode_name,
    s.sql_id,
    q.sql_text
FROM v$locked_object l
JOIN dba_objects o
  ON o.object_id = l.object_id
JOIN v$session s
  ON s.sid = l.session_id
LEFT JOIN v$sql q
  ON q.sql_id = s.sql_id
WHERE o.object_name IN (
    'INIT$_TB_FLOW_WORK',
    'INIT$_TB_FLOW_WORK_NODE',
    'INIT$_TB_FLOW_WORK_EDGE'
)
ORDER BY o.object_name, s.sid;

    waiter.serial#          AS waiting_serial,
    waiter.username         AS waiting_user,
    waiter.event            AS waiting_event,
    waiter.seconds_in_wait,
    waiter.blocking_session AS blocking_sid,

    blocker.program         AS blocking_program,
    blocker.sql_id          AS blocking_sql_id,
    q.sql_text              AS blocking_sql_text

LEFT JOIN v$session blocker
  ON blocker.sid = waiter.blocking_session
LEFT JOIN v$sql q
  ON q.sql_id = blocker.sql_id
WHERE waiter.blocking_session IS NOT NULL
ORDER BY waiter.seconds_in_wait DESC;

WITH R AS (
    SELECT S.RULE_ID
         , S.CONDITION_COUNT
         , S.RULE_CONFIDENCE
         , S.RULE_LIFT
         , S.SUPPORT_COUNT
         , S.CONDITION_TOTAL_COUNT
         , ROW_NUMBER() OVER (
               ORDER BY S.RULE_CONFIDENCE DESC NULLS LAST
                      , S.RULE_LIFT DESC NULLS LAST
                      , S.SUPPORT_COUNT DESC NULLS LAST
                      , S.RULE_ID
           ) AS DETECTION_RN
      FROM INIT$_TB_ASSOC_RULE_SUMMARY S
     WHERE S.RUN_SOURCE_TYPE = 'FLOW_WORK'
       AND S.RUN_ID = :FLOW_RUN_ID
       AND S.MODEL_NAME = 'OML_ASSOCIATION_MODEL_01'
       AND S.RESULT_HAS_VALUE_YN = 'Y'
       AND S.RESULT_COLUMN IS NOT NULL
       AND S.RESULT_VALUE IS NOT NULL
       AND S.CONDITION_TEXT IS NOT NULL
       AND NVL(S.RULE_CONFIDENCE, 0) >= 0.8
       AND NVL(S.RULE_LIFT, 0) >= 1
)
SELECT R.*
     , CASE WHEN R.DETECTION_RN <= 100 THEN '탐지대상' ELSE 'max rules 제외' END AS DETECT_STATUS
  FROM R
 WHERE R.RULE_ID = :RULE_ID;

    PROCEDURE delete_run_result_table(p_table_name IN VARCHAR2) IS
        v_sql VARCHAR2(2000);
    BEGIN
        SELECT COUNT(DISTINCT COLUMN_NAME)
          INTO v_exists
          FROM ALL_TAB_COLUMNS
         WHERE OWNER = v_owner
           AND TABLE_NAME = p_table_name
           AND COLUMN_NAME IN ('RUN_SOURCE_TYPE', 'RUN_ID');

        IF v_exists = 2 THEN
            v_sql := 'DELETE FROM "' || v_owner || '"."' || p_table_name || '"'
                  || ' WHERE "RUN_SOURCE_TYPE" = :1'
                  || '   AND "RUN_ID" = :2';
            EXECUTE IMMEDIATE v_sql USING 'FLOW_WORK', v_flow_run_id;
        END IF;
    END;

3단계 : Symbolic Regression을 통해 종속변수와 2단계에서 선택된 독립변수간의 연관규칙을 발굴
 Symbolic Regression은 파이썬 코드로 작성해서 오라클과 연동하면 될 듯 하다.

단점: 엄청난 계산량
100개의 변수로 만들 수 있는 기호의 조합은 우주에 있는 원자의 수보다 많을 수 있습니다. 
따라서 컴퓨터가 최적의 수식을 찾기 위해 수만 번 진화(Evolution) 과정을 거쳐야 하므로 계산 시간이 제법 걸립니다.

Kim의 통계적 제안:
100개의 변수를 한 번에 심볼릭 회귀에 넣으면 컴퓨터가 과부하로 멈추거나 엉뚱한 수식을 만들 확률이 높습니다.

따라서 앞서 우리가 이야기했던 '방법 2(네트워크 중심성)'나 '방법 3(랜덤 포레스트 지니 중요도)'을 통해 
가장 구조적으로 중요한 핵심 변수 5~10개만 먼저 추려낸 뒤, 
그 변수들을 대상으로 심볼릭 회귀(예: 파이썬의 PySR 라이브러리 등)를 돌려 
구체적인 공식 규칙을 뽑아내는 것이 통계학적으로 가장 완벽한 마스터플랜입니다.
