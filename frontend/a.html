function handleMenuClick(e) { e.preventDefault(); const target = e.currentTarget; const page = target.dataset.page; // [수정 핵심] 폴더 버튼(BUTTON) 클릭 시 처리 if (target.tagName === 'BUTTON') { const firstSubMenu = target.nextElementSibling?.querySelector('a'); if (firstSubMenu) { // 1. 자식 메뉴를 클릭 시뮬레이션 함 // 여기서 click()을 호출하면 handleMenuClick이 다시 실행되는데, // 그때는 target이 'A'가 되어 아래 로직을 정상적으로 탑니다. firstSubMenu.click(); return; // [중요] 폴더 버튼 자체는 여기서 실행을 끝내야 합니다. } } // 2. 제목 추출 (아이콘 제외 순수 텍스트만) const titleSpan = target.querySelector('span'); let title = titleSpan ? titleSpan.innerText.trim() : target.innerText.trim(); // 3. 홈 예외 처리 if (page === 'home' || page === 'M00000') { title = '인아이티 Data Editing 시스템'; } if (page) { // 4. 페이지 로드 및 제목 변경 실행 PageManager.load(page, title); // 5. 메뉴 활성화 스타일 처리 document.querySelectorAll('#mainNav a, #mainNav button').forEach(el => { el.classList.remove('menu-active', 'bg-blue-700'); }); // 실제 메뉴(A)일 때만 배경색 강조 if (target.tagName === 'A') { target.classList.add('menu-active', 'bg-blue-700'); } // 6. 모바일에서 사이드바 닫기 (A 태그일 때만 동작하게 됨) if (window.innerWidth <= 1024 && typeof LayoutManager !== 'undefined') { LayoutManager.close(); } } }
